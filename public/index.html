Modo Automático
Lotes eliminados: 0 
(function() {
  // CONFIGURACIÓN DE AUTO-LOOP
  const AUTO_LOOP = __AUTO_MODE_PLACEHOLDER__; 

  async function sleep(ms) {
    return new Promise(r => setTimeout(r, ms));
  }

  async function run() {
    // Selectores múltiples para robustez
    const selectors = ['div[data-latest-bg]', 'div[jsmodel]', 'c-wiz div[role="checkbox"]'];
    let photos = [];
    
    for (const selector of selectors) {
      photos = document.querySelectorAll(selector);
      if (photos.length > 0) break;
    }

    if (!photos.length) {
      console.error('No se encontraron fotos. Asegúrate de estar en photos.google.com');
      return;
    }

    console.log(`Seleccionando ${photos.length} fotos...`);
    
    // Iniciar selección
    photos[0].dispatchEvent(new MouseEvent('mouseover', {bubbles: true}));
    await sleep(500);
    photos[0].click();
    await sleep(500);

    // Seleccionar el resto
    photos.forEach((p, i) => {
      if (i > 0) p.click();
    });

    console.log('Fotos seleccionadas. Procediendo a eliminar...');
    await sleep(2000); // Retraso de 2 segundos para evitar rate limiting

    const moreBtn = document.querySelector('button[aria-label="More options"]') ||
                    document.querySelector('button[aria-label="Más opciones"]');
    if (moreBtn) moreBtn.click();
    await sleep(1000);

    const deleteBtn = Array.from(document.querySelectorAll('li[role="menuitem"]'))
      .find(el => el.textContent.includes('Delete') ||
                  el.textContent.includes('Eliminar') ||
                  el.textContent.includes('papelera'));

    if (deleteBtn) {
      deleteBtn.click();
      await sleep(1000);

      const confirmBtn = document.querySelector('button[data-mdc-dialog-action="ok"]') ||
        Array.from(document.querySelectorAll('button'))
          .find(b => b.textContent.includes('Move to trash') ||
                     b.textContent.includes('Mover a la papelera'));

      if (confirmBtn) {
        confirmBtn.click();
        
        // Incrementar contador de lotes en sessionStorage
        let batchCount = parseInt(sessionStorage.getItem('gpc_batches') || '0') + 1;
        sessionStorage.setItem('gpc_batches', batchCount);
        console.log(`Lotes eliminados: ${batchCount}`);

        await sleep(2000); // Esperar a que se procese la eliminación

        if (AUTO_LOOP) {
          console.log('Modo automático activo. Recargando en 3 segundos...');
          await sleep(3000);
          location.reload();
        } else {
          alert(`Lote de ${photos.length} fotos movido a la papelera.\nLotes eliminados en esta sesión: ${batchCount}\nRecarga Google Fotos con F5 para continuar.`);
        }
      } else {
        console.error('No se encontró el botón de confirmar.');
      }
    } else {
      console.error('No se encontró el botón de eliminar.');
    }
  }

  run();
})();
PASO 4

Repite hasta eliminar todo

El script mueve un lote a la papelera cada vez. Recarga Google Fotos con F5 y vuelve a ejecutar el script hasta que no queden fotos.

Información importante

  • Las fotos van a la papelera, no se eliminan de inmediato
  • Tienes 60 días para recuperarlas desde la papelera
  • Para eliminar definitivamente: ve a Papelera y haz click en "Vaciar papelera"
  • Si tienes miles de fotos, necesitarás ejecutar el script varias veces

Preguntas Frecuentes

¿Es seguro usar esta herramienta?

Sí, es totalmente seguro. El script se ejecuta localmente en tu navegador y no tiene acceso a tus credenciales. Solo automatiza los clics que tú harías manualmente.

¿Funciona en móvil?

No, esta herramienta requiere acceso a la consola de desarrollador del navegador, la cual solo está disponible en navegadores de escritorio (Chrome, Firefox, Edge, Safari).

¿Puedo recuperar las fotos después?

Sí, las fotos se mueven a la papelera de Google Fotos, donde permanecen por 60 días antes de ser eliminadas permanentemente, a menos que vacíes la papelera manualmente.

¿Por qué tengo que ejecutar el script varias veces?

Google Fotos solo carga una cantidad limitada de fotos a la vez en el navegador. El script selecciona lo que es visible en pantalla para asegurar que la eliminación sea exitosa sin bloquear tu cuenta.

Esta herramienta funciona completamente en tu navegador. Ningún dato es enviado a ningún servidor.
